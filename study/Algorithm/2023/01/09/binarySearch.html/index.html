int binarySearch(int data[], int n, int key) {
    int start, end;
    int mid;
 
    start = 0;
    end = n - 1;
    while (start <= end) {
        mid = (start + end) / 2;
        if (key == data[mid]) {
            return mid;        
        }
        else if (key < data[mid]) {
            end = mid - 1;
        }
        else if (key > data[mid]) {
            start = mid + 1;
        }
    }
    return -1;
}
